window.xuatExcel = function () { let table = document.querySelector("table"); // lấy bảng đang hiển thị let wb = XLSX.utils.book_new(); let ws = XLSX.utils.table_to_sheet(table); // ✅ chỉ lấy dữ liệu đã lọc // Xoá cột "Hành động" let range = XLSX.utils.decode_range(ws['!ref']); for (let R = range.s.r; R <= range.e.r; ++R) { let cell = XLSX.utils.encode_cell({ r: R, c: 6 }); // cột thứ 7 delete ws[cell]; } range.e.c = 5; // chỉ giữ 6 cột ws['!ref'] = XLSX.utils.encode_range(range); XLSX.utils.book_append_sheet(wb, ws, "Danh sách"); XLSX.writeFile(wb, "benhnhan_loc.xlsx"); }
